\usepackage{fontspec} % XeTeX
\usepackage{xunicode} % Unicode pentru XeTeX

% Limba română
\usepackage{polyglossia}
\setdefaultlanguage{romanian}

% Marginile paginii

\geometry{left=3cm}
\geometry{right=1cm}
\geometry{top=2cm}

\renewcommand{\baselinestretch}{1.4} % Distanța între rânduri de o linie și jumătate

% Fonturi, xelatex
\defaultfontfeatures{Ligatures=TeX}
\setmainfont{Tinos} % Analog pentru Times New Roman
\newfontfamily\cyrillicfont{Tinos}
\setmonofont{FreeMono} % Font pepntru codul sursă

\setmathfont{XITS Math}             % Font pentru formule: https://github.com/khaledhosny/xits-math
